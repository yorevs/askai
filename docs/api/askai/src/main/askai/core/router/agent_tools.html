 39class AgentTools(metaclass=Singleton):
 40    """This class serves as the toolkit for AskAI task agents, providing essential tools and functionalities required
 41    for their tasks.
 42    """
 43
 44    INSTANCE: "AgentTools"
 45
 46    RESERVED: list[str] = ["tools"]
 47
 48    def __init__(self):
 49        self._all: dict[str, Callable] = dict(
 50            filter(
 51                lambda pair: pair[0] not in self.RESERVED and not pair[0].startswith("_"),
 52                {n: fn for n, fn in inspect.getmembers(self, predicate=inspect.ismethod)}.items(),
 53            )
 54        )
 55
 56    @lru_cache
 57    def tools(self) -> list[BaseTool]:
 58        """Return a cached list of LangChain base tools.
 59        :return: A list of BaseTool's instances available for use.
 60        """
 61        tools: list[BaseTool] = [self._create_structured_tool(v) for _, v in self._all.items()]
 62
 63        log.debug("Available tools: are: '%s'", tools)
 64
 65        return tools
 66
 67    @property
 68    def available_tools(self) -> str:
 69        """Returns a formatted string of available tools and their descriptions.
 70        :return: A string listing tools with their descriptions, formatted for readability.
 71        """
 72        avail_list: list[str] = list()
 73        for t in self.tools():
 74            if match := re.search(r"^```(.*?)^\s*Usage:", t.description, re.DOTALL | re.MULTILINE):
 75                avail_list.append(f"**{t.name}::** " + re.sub(r"\s{2,}", " ", match.group(1).strip()))
 76        return os.linesep.join(avail_list)
 77
 78    def _human_approval(self) -> bool:
 79        """Prompt for human approval."""
 80        confirm_msg = msg.access_grant()
 81        if (resp := line_input(confirm_msg).lower()) not in ("yes", "y"):
 82            raise ValueError(f"Terminal command execution was not approved '{resp}' !")
 83        self._approved = True
 84
 85        return self._approved
 86
 87    def _create_structured_tool(self, fn: Callable) -> BaseTool:
 88        """Create a LangChain agent tool based on the provided function.
 89        :param fn: The function that implements the tool's behavior.
 90        :return: An instance of BaseTool configured with the provided function.
 91        """
 92        return StructuredTool.from_function(
 93            func=fn,
 94            name=fn.__name__,
 95            description=f"```{dedent(fn.__doc__)}```\n\n" if fn and fn.__doc__ else "",
 96            return_direct=True,
 97        )
 98
 99    def browse(self, search_query: str) -> Optional[str]:
100        """Use this tool to browse the internet for the latest news and current events, particularly when up-to-date
101        information is needed quickly. This tool is especially effective for accessing the most recent data available
102        online.
103        Usage: `browse(search_query)`
104        :param search_query: The user search query as a string. It's important to keep the user keywords as is.
105        :return: A string containing the results of the web search, or None if no relevant results are found.
106        """
107        return browse(search_query)

109    def open_url(self, url: str) -> str:
110        """Use this tool to open the browser landing at a specific website, specified by the given URL. This is not
111        useful for searching the web.
112        Usage: `open_url(url)`
113        :param url: The URL to be opened.
114        :return: A string telling whether the URL was successfully opened or not.
115        """
116        return open_url(url)

118    def query_output(self, output_query: str) -> str:
119        """Use this tool to analyze textual content, and to identify files, folders, and applications. This
120        tool is NOT designed to directly read or extract file or folder contents.
121        Usage: `query_output(output_query)`
122        :param output_query: The query regarding the output. Use "Identify <file types, names, or textual content>".
123        :return: A string containing the results of the analysis based on the query.
124        """
125        return query_output(output_query)

127    def image_captioner(self, image_path: str) -> str:
128        """Use this tool to generate a textual description of visual content, such as image files.
129        Usage: `image_captioner(image_path)`
130        :param image_path: The absolute path of the image file to be analyzed.
131        :return: A string containing the generated caption describing the image.
132        """
133        image_caption: list[str] = parse_image_caption(image_captioner(image_path))
134        return CAPTION_TEMPLATE.substitute(
135            image_path=image_path, image_caption=os.linesep.join(image_caption) if image_caption else ""
136        )

138    def webcam_capturer(self, photo_name: str | None, detect_faces: bool = False, question: str | None = None) -> str:
139        """Capture a photo via the webcam, and save it locally. Also provide a description of the objects and people
140        depicted in the picture. An additional question may address specific details regarding the photo.
141        Usage: `webcam_capturer(photo_name, detect_faces, question)`
142        :param photo_name: Optional name of the photo image file (without the extension).
143        :param detect_faces: Whether to detect and describe all faces in the photo (default is False).
144        :param question: Optional specific question about the photo taken (default is None).
145        :return: The file path of the saved JPEG image.
146        """
147        return webcam_capturer(photo_name, detect_faces, question)

153        :return: A string containing the identification and description of the person.

168        descriptive instructions to create the specified content type and can optionally save it to a file.
169        Usage: `generate_content(instructions, mime_type, filepath)`

182        :return: The absolute path name of the saved file.

198        Usage: `list_tool(folder, filters)`
199        :param folder: The absolute path of the folder whose contents you wish to list or access.
200        :param filters: Optional parameter: A comma-separated list of file globs to filter results (e.g., "*.*, *.txt").
201        :return: A string listing the contents of the folder, filtered by the provided criteria if applicable.
202        """

206        """Open and display the content of files, or playback media files, and also execute applications. This tool is

208        Usage: `open_tool(path_name)`
209        :param path_name: The absolute path of the file, folder, or application to be opened.
210        :return: The output generated by the open command, such as file content or playback status.

215        """Summarize the contents of files and folders based on user requests. This tool should be used only when the
216        user explicitly requests a summary of files and folders, not for summarizing textual content.
217        Usage: `summarize(folder, glob)`
218        :param folder: The base path of the folder containing the files to be summarized.
219        :param glob: The glob expression to specify which files should be included in the summary.
220        :return: A string containing the summary of the specified files and folders.
221        """
222        return summarize(folder, glob)

224    def terminal(self, shell_type: str, command: str) -> str:
225        """Execute terminal commands or process user-provided commands using the specified shell. This tool is used to
226        run commands in a terminal environment based on the provided shell type.
227        Usage: `terminal(shell_type, command)`
228        :param shell_type: The type of shell to use (e.g., bash, zsh, powershell, etc.).
229        :param command: The command or set of commands to execute in the terminal.
230        :return: The output of the executed terminal command(s) as a string.
231        """
232        # FIXME Check for permission before executing
233        return execute_command(shell_type, command)

239        :param reason: The reason for terminating the session.
