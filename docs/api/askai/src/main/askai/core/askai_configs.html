243    def add_device(self, device_name: str) -> None:
244        """Add a new device to the configuration list.
245        :param device_name: The device name to be added.
246        """
247        if device_name not in self._recorder_devices:
248            self._recorder_devices.add(device_name)
249            settings.put("askai.recorder.devices", ", ".join(self._recorder_devices))
250
251    def remove_device(self, device_name: str) -> None:
252        """Remove a new device from the configuration list.
253        :param device_name: The device name to be removed.
254        """
255        if device_name in self._recorder_devices:
256            self._recorder_devices.remove(device_name)
257            settings.put("askai.recorder.devices", ", ".join(self._recorder_devices))
258
259    def clear_devices(self) -> None:
260        """Remove all devices from the configuration list."""
261        self._recorder_devices.clear()
