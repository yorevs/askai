 60    @classmethod
 61    def get_device_list(cls) -> list[InputDevice]:
 62        """Return a list of available audio input devices.
 63        :return: A list of InputDevice objects representing the available microphone devices.

 65        devices = list()

 70    @staticmethod
 71    def __device_watcher() -> None:
 72        """Monitor audio input devices for being plugged in or unplugged. This method periodically checks the status of
 73        audio input devices to detect any changes.
 74        """
 75        if recorder.is_auto_swap and recorder.devices:
 76            new_list = recorder.get_device_list()
 77            new_list.sort(key=operator.itemgetter(1))
 78            if len(recorder.devices) < len(new_list):
 79                new_devices = [dev for dev in new_list if dev not in recorder.devices]
 80                for device in new_devices:
 81                    if recorder.set_device(device):
 82                        break
 83            elif len(recorder.devices) > len(new_list):
 84                for device in recorder.devices:
 85                    if recorder.set_device(device):
 86                        break
 87            recorder.devices = new_list

 89    @staticmethod
 90    def countdown(sec: int, ev: threading.Event, counter_msg: str | None) -> None:
 91        i = sec
 92        if counter_msg:
 93            events.reply.emit(reply=AIReply.mute(counter_msg + f" {i}"))
 94        while not ev.is_set() and (i := (i - 1)) >= 0:
 95            pause.milliseconds(990)
 96            if counter_msg:
 97                events.reply.emit(reply=AIReply.mute(counter_msg + f" {i}"), erase_last=True)

157    def listen(
158        self,
159        recognition_api: RecognitionApi = RecognitionApi.GOOGLE,
160        language: Language = Language.EN_US,
161        audio_path: Path | None = None,
162        raise_on_timeout: bool = True,
163        raise_on_intelligible: bool = True,
164        counter_msg: str = msg.listening(),
165    ) -> tuple[Path, Optional[str]]:
166        """Listen to the microphone, save the recorded audio as a WAV file, and transcribe the speech.
167        :param recognition_api: The API to use for recognizing the speech. Defaults to GOOGLE.
168        :param language: The spoken language. Defaults to EN_US.
169        :param audio_path: The audio file destination, generated by the listening process.
170        :param raise_on_timeout: If True, raises an error when no speech is detected within the timeout period.
171        :param raise_on_intelligible: If True, raises an error when the speech is unintelligible.
172        :param counter_msg: The message to display during the listening countdown.
173        :return: A tuple containing the path to the saved audio file and the transcribed text.
174                 If transcription fails, the second element of the tuple will be None.
175        """
176        stop_event: threading.Event = threading.Event()
177        timeout: float = seconds(configs.recorder_silence_timeout_millis)
178        limit: float = seconds(configs.recorder_phrase_limit_millis)
179
180        with Microphone(device_index=self._device_index) as mic_source:
181            try:
182                stop_event.clear()
183                counter_thread = threading.Thread(target=self.countdown, args=(limit, stop_event, counter_msg))
184                audio_path = audio_path or Path(f"{REC_DIR}/askai-stt-{now_ms()}.wav")
185                self._detect_noise()
186                counter_thread.start()
187                events.listening.emit()
188                audio: AudioData = self._rec.listen(mic_source, phrase_time_limit=limit, timeout=timeout)
189                events.listening.emit(listening=False)
190                stop_event.set()
191                counter_thread.join()
192                events.reply.emit(reply=AIReply.mute(f"{counter_msg}  "), erase_last=True)
193                stt_text = self._write_audio_file(audio, audio_path, language, recognition_api)
194            except WaitTimeoutError as err:
195                log.warning("Timed out while waiting for a speech input!")
196                if raise_on_timeout:
197                    err_msg: str = msg.timeout(f"waiting for a speech input => '{err}'")
198                    events.reply.emit(reply=AIReply.error(message=err_msg), erase_last=True)
199                stt_text = None
200            except UnknownValueError as err:
201                log.warning("Speech was not intelligible!")
202                if raise_on_intelligible:
203                    err_msg: str = msg.intelligible(err)
204                    events.reply.emit(reply=AIReply.error(message=err_msg), erase_last=True)
205                stt_text = None
206            except AttributeError as err:
207                raise InvalidInputDevice(str(err)) from err
208            except RequestError as err:
209                raise InvalidRecognitionApiError(str(err)) from err
210            finally:
211                if not stop_event.is_set():
212                    stop_event.set()
213                    counter_thread.join()
214
215        return audio_path, stt_text

217    def dictate(
218        self, recognition_api: RecognitionApi = RecognitionApi.GOOGLE, language: Language = Language.EN_US
219    ) -> Optional[str]:
220        """Captures dictated speech, processes it into text, and returns the transcribed text.
221        :param recognition_api: The API to use for recognizing the speech. Defaults to GOOGLE.
222        :param language: The spoken language. Defaults to en_US.
223        :return: A string containing the dictated text. If transcription fails, None will be returned.
224        """
225        dictated_text: str = ""
226        audio_path: Path = Path(f"{REC_DIR}/askai-dictate-{now_ms()}.wav")
227
228        while True:
229            _, phrase = self.listen(recognition_api, language, audio_path, False, False, msg.dictating())
230            if phrase:
231                display_text(f" `{('…' if dictated_text else '') + phrase}`  ", markdown=True)
232                dictated_text += (". " if dictated_text else "") + phrase.capitalize()
233                if phrase.endswith((msg.t("quit"), msg.t("exit"), msg.t("bye"), msg.t("end"))):
234                    break
235            else:
236                break
237
238        return ensure_endswith(dictated_text.capitalize(), "." + os.linesep) if dictated_text else dictated_text

240    def _write_audio_file(
241        self, audio: AudioData, audio_path: AnyPath, language: Language, recognition_api: RecognitionApi
242    ) -> Optional[str]:
243        """Write the provided audio data to disk as a file and transcribe the contents into text using the specified
244        recognition API.
245        :param audio: The audio data to be saved.
246        :param audio_path: The path where the audio file will be saved.
247        :param language: The language of the spoken content in the audio.
248        :param recognition_api: The API used for speech recognition.
249        :return: The transcribed text from the audio file. If transcription fails, returns None.
250        """
251        with open(str(audio_path), "wb") as f_rec:
252            f_rec.write(audio.get_wav_data())
253            log.debug("Voice recorded and saved as %s", audio_path)
254            if api := getattr(self._rec, recognition_api.value):
255                log.debug("Recognizing voice using %s", recognition_api)
256                assert isinstance(api, Callable)
257                return api(audio, language=language.language)
258            else:
259                raise InvalidRecognitionApiError(str(recognition_api) or "<stt-error>")
260        return None

262    def test_device(self, idx: int) -> bool:
263        """Test whether the input device specified by the given index can be used for speech-to-text (STT) input.
264        :param idx: The index of the device to be tested.
265        :return: True if the device at the specified index is suitable for STT, otherwise False.
266        """
267        log.debug(f"Testing audio input device at index: '%d'", idx)
268        test_timeout, test_phrase_limit = 0.3, 0.2
269        try:
270            with Microphone(device_index=idx) as source:
271                self._rec.listen(source, timeout=test_timeout, phrase_time_limit=test_phrase_limit)
272                return True
273        except WaitTimeoutError:  # We do expect a T/O error due to lack of audio input.
274            log.info(f"Device at index: '%d' is functional!", idx)
275            return True
276        except (AssertionError, AttributeError):
277            log.warning(f"Device at index: '%d' is non functional!", idx)
278
279        return False

281    def _detect_noise(self) -> None:
282        """Detect and adjust the background noise level."""
283        with Microphone() as source:
284            try:
285                log.debug("Adjusting noise levels…")
286                duration = seconds(configs.recorder_noise_detection_duration_millis)
287                self._rec.adjust_for_ambient_noise(source, duration=duration)
288            except UnknownValueError as err:
289                err_msg: str = msg.intelligible(f"Unable to detect noise level => '{err}'")
290                log.warning("Timed out while waiting for a speech input!")
291                events.reply.emit(reply=AIReply.error(err_msg), erase_last=True)
292
293    def _select_device(self) -> None:
294        """Select a device for recording."""
295        available: list[str] = list(filter(lambda d: d, map(str.strip, configs.recorder_devices)))
296        device: InputDevice | None = None
297        devices: list[InputDevice] = self.devices
298        while devices and not device:
299            if available:
300                for dev in list(reversed(devices)):  # Reverse to get any headphone or external device
301                    if dev[1] in available and self.set_device(dev):
302                        device = dev
303                        break
304            if not device:
305                if not (device := next(devices.__iter__(), None)):
306                    display_text(f"%HOM%%ED2%Error: Unable to setup an Audio Input device!%NC%")
307                    sys.exit(ExitStatus.FAILED.val)
308            if device and not self.set_device(device):
309                devices.remove(device)
310                device = None
