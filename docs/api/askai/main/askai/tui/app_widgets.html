118class AppSettings(Static):
119    """Application DataTable Widget."""
120
121    data = Reactive("")
122
123    def __init__(self, data: list[tuple[str, ...]] = None):
124        super().__init__()
125        self.table = DataTable(zebra_stripes=True, id="settings")
126        self.data = data or list()
127
128    def compose(self) -> ComposeResult:
129        yield self.table
130
131    async def on_mount(self) -> None:
132        self.set_class(True, "-hidden")
133
134    def watch_data(self) -> None:
135        if self.data:
136            self.table.clear(True)
137            self.table.add_columns(*self.data[0][1:])
138            for i, row in enumerate(self.data[1:], start=1):
139                label = Text(str(i), style="#B0FC38 italic")
140                self.table.add_row(*row[1:], key=row[0], label=label)
141            self.refresh()
