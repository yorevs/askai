323    async def _write_markdown(self) -> None:
324        """Write buffered text to the markdown file."""
325        if len(self._display_buffer) > 0:
326            with open(self._console_path, "a", encoding=Charset.UTF_8.val) as f_console:
327                prev_text: str | None = None
328                while len(self._display_buffer) > 0:
329                    if (text := self._display_buffer.pop(0)) == prev_text:
330                        continue
331                    prev_text = text
332                    final_text: str = text_formatter.beautify(f"{ensure_endswith(text, os.linesep * 2)}")
333                    f_console.write(final_text)
334                    f_console.flush()
335                self._re_render = True

395    @work(thread=True, exclusive=True)
396    def _ask_and_reply(self, question: str) -> bool:
397        """Ask the question and provide the reply.
398        :param question: The question to ask to the AI engine.
399        """
400        status = True
401        self.enable_controls(False)
402        try:
403            if command := re.search(self.RE_ASKAI_CMD, question):
404                with StringIO() as buf, redirect_stdout(buf):
405                    args: list[str] = list(
406                        filter(lambda a: a and a != "None", re.split(r"\s", f"{command.group(1)} {command.group(2)}"))
407                    )
408                    ask_cli(args, standalone_mode=False)
409                    if output := buf.getvalue():
410                        self.display_text(f"\n```bash\n{strip_escapes(output)}\n```")
411            elif not (reply := cache.read_reply(question)):
412                log.debug('Response not found for "%s" in cache. Querying from %s.', question, self.engine.nickname())
413                self.reply(message=msg.wait())
414                if output := splitter.process(question):
415                    self.reply(output)
416            else:
417                log.debug("Reply found for '%s' in cache.", question)
418                self.reply(reply)
419        except (NotImplementedError, ImpossibleQuery) as err:
420            self.reply_error(str(err))
421        except (MaxInteractionsReached, InaccurateResponse, ValueError, AttributeError) as err:
422            self.reply_error(msg.unprocessable(str(err)))
423        except UsageError as err:
424            self.reply_error(msg.invalid_command(err))
425        except IntelligibleAudioError as err:
426            self.reply_error(msg.intelligible(err))
427        except RateLimitError:
428            self.reply_error(msg.quote_exceeded())
429            status = False
430        except TerminatingQuery:
431            status = False
432
433        self.enable_controls()
434
435        return status

396    @work(thread=True, exclusive=True)
397    def _ask_and_reply(self, question: str) -> bool:
398        """Ask the question and provide the reply.
399        :param question: The question to ask to the AI engine.
400        """
401        status = True
402        self.enable_controls(False)
403        try:
404            if command := re.search(self.RE_ASKAI_CMD, question):
405                with StringIO() as buf, redirect_stdout(buf):
406                    args: list[str] = list(
407                        filter(lambda a: a and a != "None", re.split(r"\s", f"{command.group(1)} {command.group(2)}"))
408                    )
409                    ask_cli(args, standalone_mode=False)
410                    if output := buf.getvalue():
411                        self.display_text(f"\n```bash\n{strip_escapes(output)}\n```")
412            elif not (reply := cache.read_reply(question)):
413                log.debug('Response not found for "%s" in cache. Querying from %s.', question, self.engine.nickname())
414                self.reply(message=msg.wait())
415                if output := splitter.process(question):
416                    self.reply(output)
417            else:
418                log.debug("Reply found for '%s' in cache.", question)
419                self.reply(reply)
420        except (NotImplementedError, ImpossibleQuery) as err:
421            self.reply_error(str(err))
422        except (MaxInteractionsReached, InaccurateResponse, ValueError, AttributeError) as err:
423            self.reply_error(msg.unprocessable(str(err)))
424        except UsageError as err:
425            self.reply_error(msg.invalid_command(err))
426        except IntelligibleAudioError as err:
427            self.reply_error(msg.intelligible(err))
428        except RateLimitError:
429            self.reply_error(msg.quote_exceeded())
430            status = False
431        except TerminatingQuery:
432            status = False
433
434        self.enable_controls()
435
436        return status
