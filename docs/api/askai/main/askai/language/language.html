160    @staticmethod
161    def of_locale(loc: AnyLocale) -> "Language":
162        """Create a Language object based on a locale string or tuple containing the language code and encoding.
163        :param loc: The locale to parse.
164        """
165        # Replace possible 'loc[:.]charset' values
166        loc_enc: tuple = tuple(map(
167            lambda s: re.sub(r'(/C|C/|\'|\")+', '', s), loc
168            if isinstance(loc, tuple)
169            else re.split(r"[:.]", loc)
170        ))
171        # fmt: off
172        lang = next((
173            Language[ln.upper()]
174            for ln in list(map(
175                lambda v: v.__getitem__(0), Language.values())) if ln.casefold() == loc_enc[0].casefold()
176        ), None)
177        # fmt: on
178        check_not_none(lang, f"Unable to create Language from locale: '{loc}'")
179        lang.encoding = get_or_default(loc_enc, 1, Charset.UTF_8.val)
180        return lang
181
182    def __init__(self, locale_str: str, name: str, country: str):
183        loc_enc = re.split(r"[:.]", locale_str)
184        self._locale: tuple = loc_enc[0], get_or_default(loc_enc, 1, Charset.UTF_8.val)
185        self._name: str = name
186        self._country: str = country
187        self._encoding: Charset = Charset.of_value(self._locale[1])
188        self._idiom = self._locale[0]
189        lang = locale_str.split("_")
190        self._language, self._territory = lang[0], lang[1]
191
192    def __str__(self):
193        return f"{self.name} '{self.country}' '{str(self.encoding).upper()}'"

221        self._encoding = value if isinstance(value, Charset) else Charset.of_value(value.casefold())
