155        """Run the program."""

178        """Reply API or system errors.

180        """

182        if self.is_speak:

184        else:
185            display_text(f"Error: {message}", f"{shared.nickname}: ")

187    def _cb_reply_event(self, ev: Event, error: bool = False) -> None:
188        """Callback to handle reply events.
189        :param ev: The reply event.
190        :param error: Whether the event is an error not not.
191        """
192        if error:
193            self.reply_error(ev.args.message)
194        else:
195            verbose = ev.args.verbosity.lower()
196            if verbose == "normal" or self.is_debugging:
197                if ev.args.erase_last:
198                    cursor.erase_line()
199                self.reply(ev.args.message)

201    def _splash(self) -> None:
202        """Display the AskAI splash screen."""
203        splash_interval = 1000
204        while not self._ready:
205            if not self._processing:
206                screen.clear()
207                sysout(f"%GREEN%{self.SPLASH}%NC%")
208            pause.milliseconds(splash_interval)
209        pause.milliseconds(splash_interval)
210        screen.clear()

212    def _startup(self) -> None:
213        """Initialize the application."""
214        askai_bus = AskAiEvents.get_bus(ASKAI_BUS_NAME)
215        askai_bus.subscribe(REPLY_EVENT, self._cb_reply_event)
216        askai_bus.subscribe(REPLY_ERROR_EVENT, partial(self._cb_reply_event, error=True))
217        if self.is_interactive:
218            splash_thread: Thread = Thread(daemon=True, target=self._splash)
219            splash_thread.start()
220            nltk.download("averaged_perceptron_tagger", quiet=True, download_dir=CACHE_DIR)
221            cache.set_cache_enable(self.cache_enabled)
222            cache.read_query_history()
223            player.start_delay()
224            self._ready = True
225            splash_thread.join()
226            recorder.setup()
227            display_text(self, markdown=False)
228            self.reply(msg.welcome(os.getenv("USER", "you")))
229        else:
230            recorder.setup()
231            player.start_delay()
232        log.info("AskAI is ready to use!")

234    def _prompt(self) -> None:
235        """Prompt for user interaction."""
236        while query := shared.input_text(f"{shared.username}: "):
237            if not self._ask_and_reply(query):
238                query = None
239                break
240            else:
241                cache.save_query_history()
242        if not query:
243            self.reply(msg.goodbye())
244        sysout("")

246    def _ask_and_reply(self, question: str) -> bool:
247        """Ask the question and provide the reply.
248        :param question: The question to ask to the AI engine.
249        """
250        status = True
251        try:
252            if not (reply := cache.read_reply(question)):
253                log.debug('Response not found for "%s" in cache. Querying from %s.', question, self.engine.nickname())
254                AskAiEvents.ASKAI_BUS.events.reply.emit(message=msg.wait())
255                if output := router.process(question):
256                    self.reply(output)
257            else:
258                log.debug("Reply found for '%s' in cache.", question)
259                self.reply(reply)
260        except (NotImplementedError, ImpossibleQuery) as err:
261            self.reply_error(str(err))
262        except (MaxInteractionsReached, InaccurateResponse, ValueError) as err:
263            self.reply_error(msg.unprocessable(type(err)))
264        except TerminatingQuery:
265            status = False
266
267        return status

269    def _get_query_string(self, interactive: bool, query_arg: str | list[str]) -> None:
270        """Retrieve the proper query string used in the non interactive mode.
271        :param interactive: The interactive mode.
272        :param query_arg: The query argument provided by the command line.
273        """
274        query: str = str(" ".join(query_arg) if isinstance(query_arg, list) else query_arg)
275        self._question = query
276        dir_name, file_name = PathObject.split(self._query_prompt)
277        if not interactive:
278            stdin_args = read_stdin()
279            template = PromptTemplate(
280                input_variables=["context", "question"], template=prompt.read_prompt(file_name, dir_name)
281            )
282            final_prompt = template.format(context=stdin_args, question=self._question)
283            self._query_string = final_prompt if query else stdin_args
284            self._question = self._question or self._query_string
285        else:
286            self._query_string = query
